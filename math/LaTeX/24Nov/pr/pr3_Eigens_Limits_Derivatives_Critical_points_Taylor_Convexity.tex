% 2 Limits
\begin{problem}
   Find, if exists, the limit of the sequence, as $n\to\infty$:

    \begin{enumerate}
        \item[a) ] $\dfrac{1}{n^2}$,
        \item[b) ] $\dfrac{n^2}{2-n^3}$,
        \item[c) ] $(0.99)^n$,
        \item[d) ] $(1.11)^n$,
        \item[e) ] $\sin {\pi n}$,


    \end{enumerate}
    Which of the sequences above are convergent?
\end{problem}
